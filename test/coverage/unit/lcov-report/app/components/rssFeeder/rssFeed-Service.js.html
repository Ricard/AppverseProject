'use strict';
 
var app = angular.module('RssFeedModule');
 
app.factory('rssFeedSrv', function(){
   
        this.getFeed = function(url){
            var vm = this;
             vm.readed = {};
            var feed = new google.feeds.feed(url);
            feed.setNumEntries(10);
            feed.load(function(result){
                vm.readed = result;
            });
            return vm.readed;
        }
    
});
