'use strict';
 
var app = angular.module('RssFeedModule');
 
 
app.controller('rssFeedCtrl',
         function(rssFeedSrv, $uibModal,feedResource){
    console.log("Carga correcta del Controlador");
    var vm = this;
    vm.url = "http://feeds.weblogssl.com/genbeta";
    // List of Entry Objects o Feeds
    vm.data = [];
    // Object that includes Feeder and Feeds
    vm.feed = [];
    // List of Feeds
    vm.feedsLst = [];
    vm.feedsLst = [
        {
           feedName:"Genbeta",
           feedUrl:"http://feeds.weblogssl.com/genbeta",
           feedDesc:"Software, descargas y novedades. Las mejores aplicaciones web, con los trucos más útiles y toda la información." 
        },{
           feedName:"Xataka",
           feedUrl:"http://feeds.weblogssl.com/xataka2",
           feedDesc:"Software, descargas y novedades. Las mejores aplicaciones web, con los trucos más útiles y toda la información." 
        
        }
        
    ];
    //GET LIST OF FEEDERS
    
    feedResource.get().$promise.then(function(data){
        vm.feedsLst2 = data.data;
        vm.populateData();
        console.log("llista from backand", vm.feedsLst2);
    }); 
    
    // Add New Feeder
    vm.addFeeder = function(){
        // Create a Feeder Object
        //vm.feedLst.push(feeder);
        var modalInstance = $uibModal.open({
            animation: true,
            controller: 'rssFeedModalCtrl',
            templateUrl: 'components/rssFeeder/feedAddTemplate.html'     
        });
        
        modalInstance.result.then(function(feed) {
            vm.feedsLst.push(feed);
            vm.data = [];
            vm.populateData();
        });
    }
    
    // Recover List of Feeds
    vm.populateData = function(){
        
        for (var i = 0; i< vm.feedsLst2.length; i++){
         rssFeedSrv.getFeed(vm.feedsLst2[i].feedUrl)
            .then(function(result){
            vm.feed = result.feed;
            
            console.log("in controller", result);
            for (var i = 0; i < result.feed.entries.length; i++){
                var entry = result.feed.entries[i];
                entry.feeder = result.feed.title;
                entry.publishedDate = new Date(entry.publishedDate);
                vm.data.push(entry);
            }
            console.log("data posterior al resolve", vm.data);
        }); 
        };
        
          
    };
    vm.newFeed = new feedResource();
     
});
